import { Component, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute, Router } from '@angular/router';
import { CisService } from 'src/app/models/cis-service';
import { CisServiceImpl } from 'src/app/models/cis-service-impl';
import { ConfigurationDevice } from 'src/app/models/configuration-device';
import { ConfigurationDeviceImpl } from 'src/app/models/configuration-device-impl';
import { Device } from 'src/app/models/device';
import { DeviceImpl } from 'src/app/models/device-impl';
import { DeviceType } from 'src/app/models/device-type';
import { DeviceTypeImpl } from 'src/app/models/device-type-impl';
import { DutyTitle } from 'src/app/models/duty-title';
import { DutyTitleImpl } from 'src/app/models/duty-title-impl';
import { DataService } from 'src/app/service/data.service';
import { environment } from 'src/environments/environment';

@Component({
  selector: 'app-form-device-configuration',
  templateUrl: './form-device-configuration.component.html',
  styleUrls: ['./form-device-configuration.component.css']
})
export class FormDeviceConfigurationComponent implements OnInit {
  host: string = environment.host;

  idDT: string = "";
  idDev: string = "";
  saveApp: boolean = false;

  public dutyTitle: DutyTitle = new DutyTitleImpl();
  public device: Device = new DeviceImpl();
  public devType: DeviceType = new DeviceTypeImpl();
  public service: CisService = new CisServiceImpl();

  public configDev: ConfigurationDevice = new ConfigurationDeviceImpl();

  constructor(
    private _location: Location,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private dataService: DataService
  ) { }

  ngOnInit(): void {
    this.idDT = this.activatedRoute.snapshot.params['id1'];
    this.idDev = this.activatedRoute.snapshot.params['id2'];

    this.dataService.getDutyTitle(this.idDT).subscribe((response) =>
      this.dutyTitle = this.dataService.mapearDutyTitle(response));

    this.dataService.getDevice(this.idDev).subscribe((response) => {
      this.device = this.dataService.mapearDevices(response);

      this.dataService.getDeviceTyperUrl(this.device.deviceType).subscribe((response) => {
        this.devType = this.dataService.mapearDevType(response);
      });

      this.dataService.getSevicerUrl(this.device.cisService).subscribe((response) => {
        this.service = this.dataService.mapearService(response);
      });
    });

  }

  confirm(): void {
    this.configDev.device = `${this.host}devices/${this.idDev}`;
    this.dataService.createConf(this.configDev).subscribe();
  }

  next(): void {
    this.router.navigate([`/admin/dutytitle/${this.idDT}/devices`]);
  }

  goBack(): void {
    this._location.back();
  }

}
