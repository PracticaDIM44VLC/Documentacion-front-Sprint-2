import { Component, Input, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute } from '@angular/router';
import { Location as LocationClass } from 'src/app/models/location';
import { LocationImpl } from 'src/app/models/location-impl';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { DataService } from 'src/app/service/data.service';
import { environment } from 'src/environments/environment';
import { DataPackService } from 'src/app/service/data-pack.service';
import { PackOperation } from 'src/app/models/pack-operation';
import { PackOperationImpl } from 'src/app/models/pack-operation-impl';
import { Division } from 'src/app/models/division';

@Component({
  selector: 'app-superadmin-location',
  templateUrl: './superadmin-location.component.html',
  styleUrls: ['./superadmin-location.component.css']
})
export class SuperadminLocationComponent implements OnInit {
  id: string = "";
  host: string = environment.host;

  @Input() operation: Operation = new OperationImpl();
  operationData: Operation = new OperationImpl();

  public location: LocationClass = new LocationImpl();
  // locationData: LocationClass = new LocationImpl();

  locations: LocationClass[] = [];

  // keys = Object.keys;
  // areas = AreaCP;
  // cps = CommandPost;

  pck: PackOperation = new PackOperationImpl();
  divisions: Division[] = [];

  constructor(private _location: Location, private activateRoute: ActivatedRoute, private dataService: DataService, private dataPack: DataPackService) { }

  ngOnInit(): void {
    this.id = this.activateRoute.snapshot.params['id'];

    this.dataService.getLocationsOperation(this.id).subscribe((response) =>
      this.locations = this.dataService.extraerEditLocation(response));

    this.dataService.getOperation(this.id).subscribe((response) =>
      this.operation = this.dataService.mapearOperation(response));

    this.dataPack.getPckOperationId(this.id).subscribe((response) =>
      this.pck = this.dataPack.mapearPackOperation(response));

    setTimeout(() => {
      this.dataPack.getDivisionsPckOp(this.pck.id).subscribe((response) =>
        this.divisions = this.dataService.extraerDivision(response));
    }, 200);

  }

  viewDataLoc(loc: LocationClass): void {
    this.location = loc;
  }

  onNewLocation(loc: LocationClass): void {
    loc.operation = `${this.host}operations/${this.id}`;
    this.dataService.createLoc(loc).subscribe((response) =>
      this.ngOnInit());
  }

  onUpdateLoc(loc: LocationClass): void {
    loc.operation = `${this.host}operations/${this.id}`;
    this.dataService.updateLoc(loc).subscribe();
  }

  onDeleteLoc(): void {
    this.dataService.deleteLoc(this.location.idLocation).subscribe((response) => {
      this.locations = this.locations.filter((loc) => this.location != loc)
    });
  }

  goBack(): void {
    this._location.back();
  }

}
