import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { Location } from '@angular/common';
import { DeviceType } from 'src/app/models/device-type';
import { DeviceTypeImpl } from 'src/app/models/device-type-impl';
import { DataService } from 'src/app/service/data.service';
import { HttpClient } from '@angular/common/http';

@Component({
  selector: 'app-form-superadmin-device-type',
  templateUrl: './form-superadmin-device-type.component.html',
  styleUrls: ['./form-superadmin-device-type.component.css']
})
export class FormSuperadminDeviceTypeComponent implements OnInit {
  @Input() deviceType: DeviceType = new DeviceTypeImpl();
  @Output() deviceTypeData = new EventEmitter<DeviceType>();

  @Output() updateDeviceType = new EventEmitter<DeviceType>();

  devTypes: DeviceType[] = [];

  constructor(private _location: Location, private dataService: DataService, private httpClient: HttpClient) { }

  ngOnInit(): void {
    this.dataService.getDevTypes().subscribe((response) =>
      this.devTypes = this.dataService.extraerDevTypes(response));
  }

  selectDevTp(devTp: DeviceType): void {
    this.deviceTypeData.emit(devTp);
  }

  updateDevTp(): void {
    this.updateDeviceType.emit(this.deviceType);
  }

  newDevTp(): void {
    this.dataService.createDevTyp(this.deviceType).subscribe((response) =>
      this.deviceType = this.dataService.mapearDevType(response));
  }

  genAutoDevTp(): void {
    this.httpClient.get("assets/devicetype.json").subscribe((data) =>
      this.loadDevTp(data));
  }

  loadDevTp(objetoJSON: any) {
    objetoJSON.forEach((element: any) => {
      this.dataService.createDevTyp(element).subscribe();
      this.ngOnInit();
    });
  }

  goBack() {
    this._location.back();
  }

}
