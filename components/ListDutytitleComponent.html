import { Component, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute, Router } from '@angular/router';
import { Branch } from 'src/app/models/branch';
import { BranchImpl } from 'src/app/models/branch-impl';
import { Division } from 'src/app/models/division';
import { DivisionImpl } from 'src/app/models/division-impl';
import { DutyTitle } from 'src/app/models/duty-title';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { DataService } from 'src/app/service/data.service';
import { Device } from 'src/app/models/device';

@Component({
  selector: 'app-list-dutytitle',
  templateUrl: './list-dutytitle.component.html',
  styleUrls: ['./list-dutytitle.component.css']
})
export class ListDutytitleComponent implements OnInit {
  idOp: string = "";
  idDiv: string = "";
  idBr: string = "";

  public operation: Operation = new OperationImpl();
  public division: Division = new DivisionImpl();
  public branch: Branch = new BranchImpl();

  dutytitles: DutyTitle[] = [];
  devices: Device[] = [];

  constructor(
    private _location : Location,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private dataService: DataService
  ) {}

  ngOnInit(): void {
    this.idOp = this.activatedRoute.snapshot.params['id1'];
    this.idDiv = this.activatedRoute.snapshot.params['id2'];
    this.idBr = this.activatedRoute.snapshot.params['id3'];

    this.dataService.getOperation(this.idOp).subscribe((response) => {
      this.operation = this.dataService.mapearOperation(response);
    });

    this.dataService.getDivision(this.idDiv).subscribe((response2) => {
      this.division = this.dataService.mapearDivision(response2);
    });

    this.dataService.getBranch(this.idBr).subscribe((response3) => {
      this.branch = this.dataService.mapearBranch(response3);
    });

    this.dataService.superGetDevicesBranch(this.idBr).subscribe((response4) => {
      this.devices = this.dataService.extraerDevices(response4);

      this.devices.sort(function(a, b) {
        if (a.nameDutyTitleOfThisDevice > b.nameDutyTitleOfThisDevice) {
          return 1;
        }
        if (a.nameDutyTitleOfThisDevice < b.nameDutyTitleOfThisDevice) {
          return -1;
        }

        return 0;
      });

    });

    this.dataService.getDutyTitlesBranch(this.idBr).subscribe((response5) => {
      this.dutytitles = this.dataService.extraerDutyTitle(response5);
    });

  }

  nextDev(dt: DutyTitle): void {
    this.router.navigate([`/search/operation/${this.idOp}/division/${this.idDiv}/branch/${this.idBr}/dutytitle/${dt.idDutyTitle}/devices`]);
  }

  goBack(): void {
    this._location.back();
  }

}
