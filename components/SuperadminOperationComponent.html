import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { Location } from '@angular/common';

  ngOnInit(): void {

    });

    this.dataService.getTypeOperations().subscribe((response) => {
      this.typeOperations = this.dataService.extraerTypeOperation(response);
    });
  }

  viewDataOp(op: Operation): void {
    this.operationData.emit(op);
  }

  viewDataOpDel(op: Operation): void {
    this.operationDataDel = op;
  }

  newOperation() {
    this.dataService.createOp(this.operation).subscribe((response) => {
      this.operation = this.dataService.mapearOperation(response);
      this.ngOnInit();
    });
  }

  genPckOp(): void {
    this.pckOp.nameOperationPack = "Pack Operation " + this.operation.nameOperation;
    this.pckOp.operation = `${this.host}/operations/${this.operation.idOperation}`;

    this.dataPack.createPckOp(this.pckOp).subscribe((response) => {
      this.pckOp = this.dataPack.mapearPackOperation(response);
    });

    // this.router.navigate(['/superadmin']);
  }

  onDeleteOp(): void {
    let pckOp: PackOperation = new PackOperationImpl();
    this.dataPack.getPckOperationUrl(this.operationDataDel.packOperation).subscribe((response) => {
      pckOp = this.dataPack.mapearPackOperation(response);
      this.dataPack.deletePckOp(pckOp.id).subscribe((response) => {
        this.dataService.deleteOp(this.operationDataDel.idOperation).subscribe((response) => {
          this.operations = this.operations.filter((op) => this.operationDataDel != op);
        });
      });
    });

  }

  goBack(): void {
    this._location.back();
  }
