import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { DataService } from 'src/app/service/data.service';

@Component({
  selector: 'app-admin-operation',
  templateUrl: './admin-operation.component.html',
  styleUrls: ['./admin-operation.component.css']
})
export class AdminOperationComponent implements OnInit {
  public operation: Operation = new OperationImpl();
  operations: Operation[]= [];

  // public branch: Branch = new BranchImpl();
  // branchData: Branch = new BranchImpl();
  // branches: Branch[] = [];

  constructor(private router: Router, private dataService: DataService) { }

  ngOnInit(): void {
    this.dataService.getOperations().subscribe((response) => {
      this.operations = this.dataService.extraerOperation(response);
    });

    // this.dataService.getBranches().subscribe((response) => {
    //   this.branches = this.dataService.extraerBranch(response);
    // });
  }

  nextDT(op: Operation): void {
    this.router.navigate([`/admin/operation/${op.idOperation}/branches`]);
  }

  nextLy(op: Operation): void {
    this.router.navigate([`/admin/operation/${op.idOperation}/branches`]);
  }

  nextPers(op: Operation): void {
    this.router.navigate([`/admin/operation/${op.idOperation}/personnel`]);
  }

}
