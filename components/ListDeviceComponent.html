import { Component, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute, Router } from '@angular/router';
import { Branch } from 'src/app/models/branch';
import { BranchImpl } from 'src/app/models/branch-impl';
import { Device } from 'src/app/models/device';
import { Division } from 'src/app/models/division';
import { DivisionImpl } from 'src/app/models/division-impl';
import { DutyTitle } from 'src/app/models/duty-title';
import { DutyTitleImpl } from 'src/app/models/duty-title-impl';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { DataService } from 'src/app/service/data.service';
import { DeviceImpl } from 'src/app/models/device-impl';
import { ConfigurationDevice } from 'src/app/models/configuration-device';
import { ConfigurationDeviceImpl } from 'src/app/models/configuration-device-impl';

@Component({
  selector: 'app-list-device',
  templateUrl: './list-device.component.html',
  styleUrls: ['./list-device.component.css']
})
export class ListDeviceComponent implements OnInit {
  idOp: string = "";
  idDiv: string = "";
  idBr: string = "";
  idDT: string = "";

  public operation: Operation = new OperationImpl();
  public division: Division = new DivisionImpl();
  public branch: Branch = new BranchImpl();
  public dutytitle: DutyTitle = new DutyTitleImpl();

  public deviceData: Device = new DeviceImpl();
  devices: Device[] = [];

  public configDev: ConfigurationDevice = new ConfigurationDeviceImpl();

  constructor(
    private _location : Location,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private dataService: DataService
  ) {}

  ngOnInit(): void {
    this.idOp = this.activatedRoute.snapshot.params['id1'];
    this.idDiv = this.activatedRoute.snapshot.params['id2'];
    this.idBr = this.activatedRoute.snapshot.params['id3'];
    this.idDT = this.activatedRoute.snapshot.params['id4'];

    this.dataService.getOperation(this.idOp).subscribe((response) => {
      this.operation = this.dataService.mapearOperation(response);
    });

    this.dataService.getDivision(this.idDiv).subscribe((response2) => {
      this.division = this.dataService.mapearDivision(response2);
    });

    this.dataService.getBranch(this.idBr).subscribe((response3) => {
      this.branch = this.dataService.mapearBranch(response3);
    });

    this.dataService.getDutyTitle(this.idDT).subscribe((response) =>{
      this.dutytitle = this.dataService.mapearDutyTitle(response);
    });

    this.dataService.getDevicesDT(this.idDT).subscribe((response4) => {
      this.devices = this.dataService.extraerDevices(response4);

      this.devices.sort(function(a, b) {
        if (a.nameCisServiceOfThisDevice > b.nameCisServiceOfThisDevice) {
          return 1;
        }
        if (a.nameCisServiceOfThisDevice < b.nameCisServiceOfThisDevice) {
          return -1;
        }

        return 0;
      });
    });

  }

  devData(dev: Device): void {
    this.deviceData = dev;
    this.router.navigate([`/search/operation/${this.idOp}/division/${this.idDiv}/branch/${this.idBr}/dutytitle/${this.idDT}/device/${this.deviceData.idDevice}/config`]);
  }

  goBack(): void {
    this._location.back();
  }

}
