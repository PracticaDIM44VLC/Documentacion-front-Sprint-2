import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { Location } from '@angular/common';
import { TypeOperation } from 'src/app/models/type-operation';
import { TypeOperationImpl } from 'src/app/models/type-operation-impl';
import { DataService } from 'src/app/service/data.service';
import { PackOrigin } from 'src/app/models/pack-origin';
import { PackOriginImpl } from 'src/app/models/pack-origin-impl';
import { DataPackService } from 'src/app/service/data-pack.service';
import { environment } from 'src/environments/environment';
import { Router } from '@angular/router';
import { HttpClient } from '@angular/common/http';

@Component({
  selector: 'app-form-superadmin-type-operation',
  templateUrl: './form-superadmin-type-operation.component.html',
  styleUrls: ['./form-superadmin-type-operation.component.css']
})
export class FormSuperadminTypeOperationComponent implements OnInit {
  host: string = environment.host;

  @Input() typeOperation: TypeOperation = new TypeOperationImpl();
  @Output() typeOperationData = new EventEmitter<TypeOperation>();

  public pack: PackOrigin = new PackOriginImpl();

  constructor(
    private _location: Location,
    private router: Router,
    private dataService: DataService,
    private dataPack: DataPackService,
    private httpClient: HttpClient
  ) { }

  ngOnInit(): void {}

  // selectTOp(tpOp: TypeOperation): void {
  //   this.typeOperationData.emit(tpOp);
  // }

  newTOp(): void {
    this.dataService.createTOp(this.typeOperation).subscribe((response) => {
      this.typeOperation = this.dataService.mapearTypeOperation(response);
    });
  }

  genPck(): void {
    this.pack.namePackOrigin = "Pack " + this.typeOperation.nameTypeOperation;
    this.pack.typeOperation = `${this.host}/typeoperations/${this.typeOperation.idTypeOperation}`;

    this.dataPack.createPck(this.pack).subscribe((response) => {
      this.pack = this.dataPack.mapearPackOrigin(response);
      // this.genAutoDiv();
    });

    this.router.navigate(['/superadmin/config/new-items']);

  }

  // genAutoDiv(): void {

  }

}
