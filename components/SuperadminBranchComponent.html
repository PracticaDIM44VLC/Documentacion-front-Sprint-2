import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute, Router } from '@angular/router';
import { Branch } from 'src/app/models/branch';
import { BranchImpl } from 'src/app/models/branch-impl';
import { Division } from 'src/app/models/division';
import { DivisionImpl } from 'src/app/models/division-impl';
import { DataService } from 'src/app/service/data.service';
import { environment } from 'src/environments/environment';
import { HttpClient } from '@angular/common/http';
import { DataPackService } from 'src/app/service/data-pack.service';
import { PackOrigin } from 'src/app/models/pack-origin';
import { PackOriginImpl } from 'src/app/models/pack-origin-impl';

@Component({
  selector: 'app-superadmin-branch',
  templateUrl: './superadmin-branch.component.html',
  styleUrls: ['./superadmin-branch.component.css']
})
export class SuperadminBranchComponent implements OnInit {
  id: string = "";
  host: string = environment.host;

  @Input() division: Division = new DivisionImpl();

  public branch: Branch = new BranchImpl();
  branchData: Branch = new BranchImpl();
  branchDataDel: Branch = new BranchImpl();
  @Output() branchDataUp = new EventEmitter<Branch>();
  branches: Branch[] = [];

  public pckOrg: PackOrigin = new PackOriginImpl();

  constructor(
    private _location: Location,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private dataService: DataService,
    private dataPack: DataPackService,
    private httpClient: HttpClient) { }

  ngOnInit(): void {
    this.id = this.activatedRoute.snapshot.params['id'];

    this.dataService.getDivision(this.id).subscribe((response) => {
      this.division = this.dataService.mapearDivision(response);
      this.dataPack.getPckOriginUrl(this.division.packOrigin).subscribe((response) => {
        this.pckOrg = this.dataPack.mapearPackOrigin(response);
      });
    });

    this.dataService.getBranchesDivision(this.id).subscribe((response) =>
      this.branches = this.dataService.extraerEditBranch(response));
  }

  viewDataUp(br: Branch): void {
    this.branchDataUp.emit(br);
  }

  viewDataDel(br: Branch): void {
    this.branchDataDel = br;
  }

  genBr(): void {
    this.httpClient.get("assets/branches.json").subscribe((data) => {
      this.loadBranch(data);
    });
  }

  loadBranch(objetoJSON: any) {
    objetoJSON.divisionsOfThisOperation.forEach((element: any) => {
      if (element.nameDivision == this.division.nameDivision) {
        element.branchesOfThisDivision.forEach((element2: any) => {
          element2.division = `${this.host}divisions/${this.id}`;
          this.dataService.createBr(element2).subscribe((response) =>
            this.ngOnInit()
          );
        });
      }
    });

  }

  newBranch(br: Branch): void {
    br.division = `${this.host}/divisions/${this.division.idDivision}`;
    this.dataService.createBr(br).subscribe((response) =>
      this.ngOnInit()
    );
  }

  onDeleteBr(): void {
    this.dataService.deleteBr(this.branchDataDel.idBranch).subscribe((response) => {
      this.branches = this.branches.filter((br) => this.branchDataDel != br)
    });
  }

  goBack(): void {
    this._location.back();
  }

}
