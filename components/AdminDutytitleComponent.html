import { Component, Input, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute, Router } from '@angular/router';
import { Branch } from 'src/app/models/branch';
import { BranchImpl } from 'src/app/models/branch-impl';
import { DutyTitle } from 'src/app/models/duty-title';
import { DutyTitleImpl } from 'src/app/models/duty-title-impl';
import { DataService } from 'src/app/service/data.service';
import { WorkStation } from 'src/app/models/work-station';
import { WorkStationImpl } from 'src/app/models/work-station-impl';

@Component({
  selector: 'app-admin-dutytitle',
  templateUrl: './admin-dutytitle.component.html',
  styleUrls: ['./admin-dutytitle.component.css']
})
export class AdminDutytitleComponent implements OnInit {
  id: string = "";

  public dutytitle: DutyTitle = new DutyTitleImpl();
  dutytitles: DutyTitle[] = [];

  public branch: Branch = new BranchImpl();

  // public workStation: WorkStation = new WorkStationImpl();
  workStations: WorkStation[] = [];

  constructor(
    private _location: Location,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private dataService: DataService
  ) { }

  ngOnInit(): void {
    this.id = this.activatedRoute.snapshot.params['id'];

    this.dataService.getBranch(this.id).subscribe((response) => {
      this.branch = this.dataService.mapearBranch(response);
    });

    this.dataService.getDutyTitlesBranch(this.id).subscribe((response) => {
      this.dutytitles = this.dataService.extraerDutyTitle(response);
    });

    // this.dataService.getWorkStations().subscribe((response) => {
    //   this.workStations = this.dataService.extraerWS(response);
    // });
  }

  goBack(): void {
    this._location.back();
  }

}
