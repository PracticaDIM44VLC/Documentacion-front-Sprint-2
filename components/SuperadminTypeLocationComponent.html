import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { Location } from '@angular/common';
import { TypeLocation } from 'src/app/models/type-location';
import { TypeLocationImpl } from 'src/app/models/type-location-impl';
import { DataService } from 'src/app/service/data.service';
import { HttpClient } from '@angular/common/http';

@Component({
  selector: 'app-superadmin-type-location',
  templateUrl: './superadmin-type-location.component.html',
  styleUrls: ['./superadmin-type-location.component.css']
})
export class SuperadminTypeLocationComponent implements OnInit {
  @Input() typeLocation: TypeLocation = new TypeLocationImpl();
  @Output() typeLocationData = new EventEmitter<TypeLocationImpl>();

  public typeLocationDataDel: TypeLocation = new TypeLocationImpl();

  typeLocations: TypeLocation[] = [];

  constructor(private _location: Location, private dataService: DataService, private httpClient: HttpClient) { }

  ngOnInit(): void {
    this.dataService.getTypeLocations().subscribe((response) => {
      this.typeLocations = this.dataService.extraerTypeLocation(response);
    });
  }

  viewDataTLoc(typeLocation: TypeLocation): void {
    this.typeLocationData.emit(typeLocation);
  }

  viewDataDelTLoc(typeLocation: TypeLocation): void {
    this.typeLocationDataDel = typeLocation;
  }

  // genAutoTLoc(): void {
  //   this.httpClient.get("assets/typelocation.json").subscribe((data) =>
  //     this.loadTypeLocation(data));
  // }

  // loadTypeLocation(objetoJSON: any) {
  //   objetoJSON.forEach((element: any) => {
  //     this.dataService.createTpLoc(element).subscribe();
  //     this.ngOnInit();
  //   });

  // }

  onDeleteTLoc(): void {
    this.dataService.deleteTLoc(this.typeLocationDataDel.idTypeLocation).subscribe((response) => {
      this.typeLocations = this.typeLocations.filter((tLoc) => this.typeLocationDataDel != tLoc);
    });
  }

  goBack(): void {
    this._location.back();
  }

}
