import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import html2canvas from 'html2canvas';
import jsPDF from 'jspdf';
import { Branch } from 'src/app/models/branch';
import { CisService } from 'src/app/models/cis-service';
import { Device } from 'src/app/models/device';
import { DeviceType } from 'src/app/models/device-type';
import { Division } from 'src/app/models/division';
import { DutyTitle } from 'src/app/models/duty-title';
import { Location as LocationClass } from 'src/app/models/location';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { PackOperation } from 'src/app/models/pack-operation';
import { PackOperationImpl } from 'src/app/models/pack-operation-impl';
import { DataPackService } from 'src/app/service/data-pack.service';
import { DataService } from 'src/app/service/data.service';

@Component({
  selector: 'app-report',
  templateUrl: './report.component.html',
  styleUrls: ['./report.component.css'],
})
export class ReportComponent implements OnInit {
  idOp: string = "";

  div = false;
  br = false;
  dt = false;
  dev = false;
  loc = false;

  acum: number = 0;

  public operation: Operation = new OperationImpl();
  public pckOp: PackOperation = new PackOperationImpl();

  divisions: Division[] = [];
  branches: Branch[] = [];
  dutytitles: DutyTitle[] = [];
  devices: Device[] = [];
  locations: LocationClass[] = [];

  services: CisService[] = [];
  devTypes: DeviceType[] = [];

  constructor(
    private activatedRoute: ActivatedRoute,
    private dataService: DataService,
    private dataPack: DataPackService
  ) {}

  ngOnInit(): void {
    this.idOp = this.activatedRoute.snapshot.params['id'];

    this.dataService.getOperation(this.idOp).subscribe((response) => {
      this.operation = this.dataService.mapearOperation(response);

      this.dataPack.getPckOperationUrl(this.operation.packOperation).subscribe((response2) => {
        this.pckOp = this.dataPack.mapearPackOperation(response2);

        this.dataPack.getDivisionsPckOp(this.pckOp.id).subscribe((response3) => {
          this.divisions = this.dataService.extraerDivision(response3);
        });
      });
    });

    this.dataService.superGetBranch(this.idOp).subscribe((response) => {
      this.branches = this.dataService.extraerBranch(response);
    });

    this.dataService.superGetDTOp(this.idOp).subscribe((response) => {
      this.dutytitles = this.dataService.extraerDutyTitle(response);
    });

    this.dataService.superGetDevicesOp(this.idOp).subscribe((response) => {
      this.devices = this.dataService.extraerDevices(response);
    });

    this.dataService.getLocationsOperation(this.idOp).subscribe((response) => {
      this.locations = this.dataService.extraerLocation(response);
    });

    this.dataService.getServices().subscribe((response => {
      this.services = this.dataService.extraerServices(response);
    }));

    this.dataService.getDevTypes().subscribe((response => {
      this.devTypes = this.dataService.extraerDevTypes(response);
    }));

  }

  divs(): void {
    this.div = !this.div;
  }
  brs(): void {
    this.br = !this.br;
  }
  dts(): void {
    this.dt = !this.dt;
  }
  devs(): void {
    this.dev = !this.dev;
  }
  locs(): void {
    this.loc = !this.loc;
  }

  reportPDF(): void {
    let DATA: any = document.getElementById('htmlData');
    html2canvas(DATA).then((canvas) => {
      let fileWidth = 208;
      let fileHeight = (canvas.height * fileWidth) / canvas.width;
      const FILEURI = canvas.toDataURL('image/png');
      let PDF = new jsPDF('p', 'mm', 'a4');
      let position = 0;
      PDF.addImage(FILEURI, 'PNG', 0, position, fileWidth, fileHeight);
      PDF.save(`layout_${this.operation.nameOperation}.pdf`);
    });

  }

}
