import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { DutyTitle } from 'src/app/models/duty-title';
import { DutyTitleImpl } from 'src/app/models/duty-title-impl';
import { Military } from 'src/app/models/military';
import { MilitaryImpl } from 'src/app/models/military-impl';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { DataService } from 'src/app/service/data.service';

@Component({
  selector: 'app-admin-user',
  templateUrl: './admin-user.component.html',
  styleUrls: ['./admin-user.component.css']
})
export class AdminUserComponent implements OnInit {
  id: string = "";

  public operation: Operation = new OperationImpl();

  public dutyTitle: DutyTitle = new DutyTitleImpl();
  public dutyTitleData: DutyTitle = new DutyTitleImpl();
  public dutyTitleDataUpdate: DutyTitle = new DutyTitleImpl();
  dutytitles: DutyTitle[] = [];

  public user: Military = new MilitaryImpl();
  public userData: Military = new MilitaryImpl();

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private dataService: DataService
  ) { }

  ngOnInit(): void {
    this.id = this.activatedRoute.snapshot.params['id'];

    this.dataService.getOperation(this.id).subscribe((response) => {
      this.operation = this.dataService.mapearOperation(response);
    });

    this.dataService.superGetDTOp(this.id).subscribe((response) => {
      this.dutytitles = this.dataService.extraerDutyTitle(response);

      this.dutytitles.sort(function(a, b) {
        if (a.nameDutyTitle > b.nameDutyTitle) {
          return 1;
        }
        if (a.nameDutyTitle < b.nameDutyTitle) {
          return -1;
        }

        return 0;
      });
    });
  }

  viewData(dt: DutyTitle): void {
    this.dutyTitleData = dt;
    this.router.navigate([`/admin/pers/dutytitle/${this.dutyTitleData.idDutyTitle}`]);
  }

  viewDataUp(dt: DutyTitle): void {
    this.dutyTitleDataUpdate = dt;
  }

  viewDataUser(usu: Military): void {
    this.userData = usu;
  }

  onUpdateUser(usu: Military): void {
    this.dataService.updateMilitary(usu).subscribe();
  }

}
