import { Component, Input, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { Device } from 'src/app/models/device';
import { DutyTitle } from 'src/app/models/duty-title';
import { DutyTitleImpl } from 'src/app/models/duty-title-impl';
import { DeviceImpl } from 'src/app/models/device-impl';
import { ActivatedRoute, Router } from '@angular/router';
import { DataService } from 'src/app/service/data.service';
import { DeviceType } from 'src/app/models/device-type';
import { CisService } from 'src/app/models/cis-service';
import { environment } from 'src/environments/environment';

@Component({
  selector: 'app-admin-device',
  templateUrl: './admin-device.component.html',
  styleUrls: ['./admin-device.component.css']
})
export class AdminDeviceComponent implements OnInit {
  id: string = "";
  host: string = environment.host;

  @Input() dutytitle: DutyTitle = new DutyTitleImpl();

  public device: Device = new DeviceImpl()
  devices: Device[] = [];

  deviceTypes: DeviceType[] = [];
  cisServices: CisService[] = [];

  constructor(
    private _location: Location,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private dataService: DataService
  ) { }

  ngOnInit(): void {
    this.id = this.activatedRoute.snapshot.params['id'];

    this.dataService.getDutyTitle(this.id).subscribe((response) => {
      this.dutytitle = this.dataService.mapearDutyTitle(response);
    });
    this.dataService.getDevTypes().subscribe((response) => {
      this.deviceTypes = this.dataService.extraerDevTypes(response);
    });
    this.dataService.getServices().subscribe((response) => {
      this.cisServices = this.dataService.extraerServices(response);
    });

    this.dataService.getDevicesDT(this.id).subscribe((response) => {
      this.devices = this.dataService.extraerDevices(response);
    });

  }

  confirm(): void {
    this.router.navigate([`/admin/branch/${this.dutytitle.idBranchOfThisDutyTitle}/dutytitles`]);
  }

  onNewDevice(dev: Device): void {
    dev.dutyTitle = `${this.host}dutytitles/${this.id}`;
    this.dataService.createDev(dev).subscribe((response) => {
      this.ngOnInit();
    });
  }

  onDeleteDev(dev: Device): void {

  }

  goBack(): void {
    this._location.back();
  }

}
