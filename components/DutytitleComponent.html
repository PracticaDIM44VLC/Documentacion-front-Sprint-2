import { DutyTitle } from 'src/app/models/duty-title';
import { DutyTitleImpl } from 'src/app/models/duty-title-impl';
import { DataService } from 'src/app/service/data.service';

@Component({
  selector: 'app-dutytitle',
  templateUrl: './dutytitle.component.html',
  styleUrls: ['./dutytitle.component.css']
})
export class DutytitleComponent implements OnInit {
  @Input() dutytitle: DutyTitle = new DutyTitleImpl();
  @Output() dutytitleData = new EventEmitter<DutyTitle>();

  constructor(private dataService: DataService) {}

  ngOnInit(): void {
  }

  dataDT(): void {
    this.dutytitleData.emit(this.dutytitle);
  }

}
