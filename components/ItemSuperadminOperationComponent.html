import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { faPenToSquare, faTrashCan } from '@fortawesome/free-solid-svg-icons';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { TypeOperation } from 'src/app/models/type-operation';
import { TypeOperationImpl } from 'src/app/models/type-operation-impl';
import { DataService } from 'src/app/service/data.service';

@Component({
  selector: 'app-item-superadmin-operation',
  templateUrl: './item-superadmin-operation.component.html',
  styleUrls: ['./item-superadmin-operation.component.css'],
})
export class ItemSuperadminOperationComponent implements OnInit {
  // inboundClick = false;
  url: string = "";
  param: string = "";

  trash = faTrashCan;
  edit = faPenToSquare;

  @Input() operation: Operation = new OperationImpl();
  @Output() operationUpdate = new EventEmitter<Operation>();
  @Output() operationDelete = new EventEmitter<Operation>();

  public typeOperation: TypeOperation = new TypeOperationImpl();

  constructor(private activatedRoute: ActivatedRoute, private router: Router, private dataService: DataService) {}

  ngOnInit(): void {
    this.url = this.activatedRoute.snapshot.toString();
    this.param = this.dataService.getParam(this.url);

    this.dataService.getTOpOperation(this.operation.idOperation).subscribe((response) =>
      this.typeOperation = this.dataService.mapearTypeOperation(response)
    );
  }

  // toggleData(): void {
  //   this.inboundClick = !this.inboundClick;
  // }

  structureUpdate(): void {
    this.router.navigate([`/superadmin/config/update/operation/${this.operation.idOperation}/divisions`]);
  }

  infrastructureUpdate(): void {
    this.router.navigate([`/superadmin/config/update/operation/${this.operation.idOperation}/locations`]);
  }

  structureDelete(): void {
    this.router.navigate([`/superadmin/config/delete/operation/${this.operation.idOperation}/divisions`]);
  }

  infrastructureDelete(): void {
    this.router.navigate([`/superadmin/config/delete/operation/${this.operation.idOperation}/locations`]);
  }



  updateOp(): void {
    this.operationUpdate.emit(this.operation);
  }

  deleteOp(): void {
    this.operationDelete.emit(this.operation);
  }

}
