import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';
import { ListBranchComponent } from './search-branch/list-branch/list-branch.component';
import { DeviceConfigComponent } from './search-device/device-config/device-config.component';
import { ListDeviceComponent } from './search-device/list-device/list-device.component';
import { ListDivisionComponent } from './search-division/list-division/list-division.component';
import { ListDutytitleComponent } from './search-dutytitle/list-dutytitle/list-dutytitle.component';
import { SearchHomeComponent } from './search-home/search-home.component';
import { ReportComponent } from './search-operation/report/report.component';
import { SearchOperationComponent } from './search-operation/search-operation.component';
import { SearchShellComponent } from './search-shell/search-shell.component';

const routes: Routes = [
  {
    path: '',
    component: SearchShellComponent,
    children: [
      {
        path: '',
        component: SearchHomeComponent
      },
      {
        path: 'operations',
        component: SearchOperationComponent
      },
      {
        path: 'operation/:id/divisions',
        component: ListDivisionComponent
      },
      {
        path: 'operation/:id1/division/:id2/branches',
        component: ListBranchComponent
      },
      {
        path: 'operation/:id1/division/:id2/branch/:id3/dutytitle',
        component: ListDutytitleComponent
      },
      {
        path: 'operation/:id1/division/:id2/branch/:id3/dutytitle/:id4/devices',
        component: ListDeviceComponent
      },
      {
        path: 'operation/:id1/division/:id2/branch/:id3/dutytitle/:id4/device/:id5/config',
        component: DeviceConfigComponent
      },
      {
        path: 'operation/:id/report',
        component: ReportComponent
      }
    ]
  }
];

@NgModule({
  imports: [RouterModule.forChild(routes)],
  exports: [RouterModule]
})
export class SearchRoutingModule { }
