import { HttpClient, HttpErrorResponse, HttpStatusCode } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { catchError, Observable, throwError } from 'rxjs';
import { environment } from 'src/environments/environment';
import { PackOperation } from '../models/pack-operation';
import { PackOperationImpl } from '../models/pack-operation-impl';
import { PackOrigin } from '../models/pack-origin';
import { PackOriginImpl } from '../models/pack-origin-impl';

@Injectable({
  providedIn: 'root'
})
export class DataPackService {
  // url de la API
  private host: string = environment.host;
  // url de Pack Origin
  private urlPck: string = `${this.host}packorigins`;
  // url de Pack Operation
  private urlPckOp: string = `${this.host}operationpacks`;
  // url de Operation
  private urlOp: string = `${this.host}operations`;

  constructor(private http: HttpClient) { }

  // método para conseguir el ID
  getId(url: string): string {
    let posicionFinal: number = url.lastIndexOf('/');
    let numId: string = url.slice(posicionFinal + 1, url.length);

    return numId;
  }

  /*
   * Lectura de datos
   */
  getPckOriginUrl(url: string): Observable<PackOrigin> {
    return this.http.get<PackOrigin>(`${url}`)
      // .pipe(
      //   catchError((err: HttpErrorResponse) => {
      //     return this.handleErrors(err)
      //   })
      // )
      .pipe(
        catchError((err) => {
          console.log(`No Pack Origin: ${err.error}`);
          return throwError(err);
        })
      )
  }
  // handleErrors(error: HttpErrorResponse): Observable<never>  {
  //   if (error.status == HttpStatusCode.Forbidden)
  //     return throwError('Request forbidden');
  //   if (error.status == HttpStatusCode.NotFound)
  //     return throwError('No Pack Origin asociated');
  //   if (error.status == HttpStatusCode.InternalServerError)
  //     return throwError('Error in the server');
  //   return throwError('An error has ocurred');
  // }

  getPckOperationUrl(url: string): Observable<PackOperation> {
    return this.http.get<PackOperation>(`${url}`);
  }
  getPckOperationId(id: string): Observable<PackOperation> {
    return this.http.get<PackOperation>(`${this.urlOp}/${id}/operationPack`);
  }

  getDivisionsPckOp(id: string): Observable<any> {
    return this.http.get<any>(`${this.urlPckOp}/${id}/divisionsOfThisOperationPack`);
  }

  /*
   * Creación de datos
   */
  createPck(pack: PackOrigin): Observable<any> {
    return this.http.post(this.urlPck, pack);
  }

  createPckOp(packOp: PackOperation): Observable<any> {
    return this.http.post(this.urlPckOp, packOp);
  }


  /*
   * Borrado de datos
   */
  deletePckOrg(id: string): Observable<PackOrigin> {
    return this.http.delete<PackOrigin>(`${this.urlPck}/${id}`);
  }

  deletePckOp(id: string): Observable<PackOperation> {
    return this.http.delete<PackOperation>(`${this.urlPckOp}/${id}`);
  }

  /*
   * Mapear datos Pack
   */
  extraerPackOrigin(respuestaApi: any): PackOrigin[] {
    const packs: PackOrigin[] = [];
    let respuesta: any = respuestaApi._embedded.packs;
    if (respuesta === undefined) {
      console.info("No Pack");
    } else
    respuestaApi._embedded.packs.forEach((pck: any) => {
      packs.push(this.mapearPackOrigin(pck));
    });

    return packs;
  }
  mapearPackOrigin(packApi: any): PackOrigin {
    let pack: PackOrigin = new PackOriginImpl();
    pack.idPackOrigin = this.getId(packApi._links.self.href);
    pack.namePackOrigin = packApi.namePackOrigin;
    pack.divisionsOfThisPack = packApi._links.divisionsOfThisPack.href;
    pack.typeOperation = packApi._links.typeOperation.href;

    return pack;
  }

  /*
   * Mapear datos Pack Operation
   */
  extraerPackOperation(respuestaApi: any): PackOperation[] {
    const packOps: PackOperation[] = [];
    let respuesta: any = respuestaApi._embedded.operationpacks;
    if (respuesta === undefined) {
      console.info("No Pack Operation");
    } else
    respuestaApi._embedded.operationpacks.forEach((pckOp: any) => {
      packOps.push(this.mapearPackOperation(pckOp));
    });

    return packOps;
  }
  mapearPackOperation(packOpApi: any): PackOperation {
    let packOp: PackOperation = new PackOperationImpl();
    packOp.id = this.getId(packOpApi._links.operationpacks.href);
    packOp.nameOperationPack = packOpApi.nameOperationPack;
    packOp.divisionsOfThisOperationPack = packOpApi._links.divisionsOfThisOperationPack.href;
    packOp.operation = packOpApi._links.operation.href;

    return packOp;
  }

}
